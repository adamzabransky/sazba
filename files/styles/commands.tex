\newcommand\NiceField[2][]{%
\raisebox{-2pt}{\TextField[#1,borderwidth=0pt,bordercolor=0.99 0.99 0.99]{\raisebox{2pt}{#2 }}}%
}

\newlength\TextFieldLength
\newcommand\NiceFieldFill[2][]{% definice formuláře, který se automaticky roztahuje
  \setlength\TextFieldLength{\linewidth}%
  \settowidth{\dimen0}{#2 }%
  \addtolength\TextFieldLength{-\dimen0}%
  \addtolength\TextFieldLength{-3.0pt}%
  \NiceField[#1,width=\TextFieldLength]{#2}%
}
